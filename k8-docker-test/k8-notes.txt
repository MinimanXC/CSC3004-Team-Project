**Docker**

- Server
$ docker image rm bc-server
$ docker build -t bc-server .
$ docker tag bc-server:latest syabz03/bc-server:latest

- Client
$ docker image rm bc-client
$ docker build -t bc-client .
$ docker run -p 4444:4444 -d bc-client
$ docker tag bc-client:latest syabz03/bc-client:latest

**Kubernetes**

Add Namespace 
$ kubectl create namespace bc-server

PV/PVC
kubectl create -f persistent_volume.yml
kubectl create -f persistent_volume_claim.yml

Deployment/Services
kubectl create -f bc-server.deployment.yml 
kubectl create -f bc-server.service.yml

Get Services 
$ kubectl get all -n bc-server
$ kubectl get pv,pvc -n bc-server 
$ kubectl get services -n bc-server -o wide
$ kubectl get pods --all-namespaces
$ kubectl describe pod <pod name> -n bc-server
$ kubectl logs <pod name> -n bc-server

* Delete Everything
kubectl delete --all pods --namespace=bc-server
kubectl delete --all deployments --namespace=bc-server
kubectl delete --all services --namespace=bc-server
